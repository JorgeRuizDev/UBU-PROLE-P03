| <IF: "IF">
| <ELSE: "ELSE">
| <ENDIF: "END-IF">
| <WHEN: "WHEN">
| <COMPUTE: "COMPUTE">
| <MOVE: "MOVE">
| <EVALUATE: "EVALUATE">
| <ENDEVAL: "END-EVALUATE">
| <PERFORM: "PERFORM">
| <ENDPERF: "END-PERFORM">

| <DISPLAY: "DISPLAY">
| <TO: "TO">

NON-TERMINALS

/**
* Axiom
*/
Run ::= Sentences <EOF> 
/**
* Sentences Terminal
*/
Sentences ::= ( Sent )+ 
/**
* Sentences'
*/
Sent ::= Assig
| Proc 
/**
* Assign Operators
*/
Assig ::= <COMPUTE> <ID> <EQUALS> Arithexp
| <MOVE> ( <NUM> | <ID> ) <TO> <ID> 
/**
* Statments
*/
Proc ::= <IF> Arithexp Sentences Elseopt
| <EVALUATE> <ID> ( WhenClause )+ <ENDEVAL>
| <PERFORM> <UNTIL> Arithexp Sentences <ENDPERF>
| <DISPLAY> Arithexp 
/**
* Else and End if
*/
Elseopt ::= <ELSE> Sentences <ENDIF>
| <ENDIF> 
/**
* When Clause (Evaluate sub-statement)
* @Param id (String): The current ID that is being evaluated.
* @Param end_label: The last statement.
*/
WhenClause ::= <WHEN> Arithexp Sentences 
/**
* Arithmetic Expresion
*/
Arithexp ::= Multexp ArithexpP 
/**
* Recursive Arithmetic Expresion
*/
ArithexpP ::= ( <ADD> Multexp ArithexpP | <SUB> Multexp ArithexpP )? 
/**
* Arithmetic Expresion (Values and * or /)
*/
Multexp ::= Value MultexpP 
/**
* Multiplication Expresion (Recursive)
*/
MultexpP ::= ( <MULT> Value MultexpP | <DIV> Value MultexpP )? 
/**
* A Value
* Recognizes:
* - An INT
* - A String that starts with a letter
* - An arithmetic expresion.
*/
Value ::= <NUM>
| <ID>
| "(" Arithexp ")"
